  colframe=Salmon,
  colback=white,
  frame code app={
    \fill[rounded corners,Salmon] ([yshift=-.5cm]frame.north west)rectangle([yshift=32pt]frame.north east);

      \foreach \x in {-11,-7,...,11}{%
        \fill ([xshift=\x mm]a) circle (2.5pt);
        \node[
          inner sep=0pt, outer sep=0pt, 
          minimum height=1cm, minimum width=3.5pt,
          bottom color=black!70!white, top color=black!20!white,
          anchor=south, rounded corners=0.5pt
        ] at ([xshift=\x mm]a) {};
      }
      \advance\tmp@length by 0.25\textwidth\relax%
    }
  }
}
